/**
 * ====================================================================
 * ##　Memo
 * ①TSファイルのコンパイルコマンド
 *  tsc main.ts
 * ②コンパイルJSの実行コマンド
 *  node main.js
 * ③標準入力の受け取り
 * const input = require('fs').readFileSync('/dev/stdin', 'utf8');
 * ④標準入力の終了
 *  Ctrl + D
 * 
 * ## 課題
 * URL : https://judge.u-aizu.ac.jp/onlinejudge/description.jsp?id=ALDS1_5_A&lang=ja
 * 
 * ## 問題概要
 * 長さ n の数列 A と整数 m に対して、A の要素の中のいくつかの要素を足し合わせて m が作れるかどうかを判定するプログラムを作成してください。
 * A の各要素は1度だけ使用できます。
 * 
 * 数列 A が与えられたうえで、質問として q 個の mi が与えられるので、それぞれについて "yes" または "no" と出力してください。
 * 
 * ## 入力
 * 1. 1行目に整数 n（数列の長さ）
 * 2. 2行目に数列 A を表す n 個の整数
 * 3. 3行目に整数 q（質問の数）
 * 4. 4行目に q 個の整数 mi（質問の値）
 * 
 * ## 出力
 * 各質問について、A の要素を足し合わせて mi を作ることができれば "yes"、できなければ "no" を出力してください。
 * 
 * ## 制約
 * - n ≤ 20
 * - q ≤ 200
 * - 1 ≤ A の要素 ≤ 2,000
 * 
 * ====================================================================
 */

const main = () =>{
    const input: string = require('fs').readFileSync('/dev/stdin', 'utf8');
    const lines:string[] = input.split('\n');
    
    try{
        // step1 Aの要素数nを受け取る
        const n: number = parseInt(lines[0].trim())
        // step2 Aの要素を受け取る
        const A: number[] = lines[1].trim().split(' ').map((str: string) => parseInt(str));

        checkCorrectArray(A, n);
        // step3 mの要素数qを受け取る
        const q: number = parseInt(lines[2].trim())
        // step4 mの要素を受け取る
        const m: number[] = lines[3].trim().split(' ').map((str: string) => parseInt(str));

        checkCorrectArray(m, q);

        exhaustiveSearch(m, A, n);

    }catch(e){
        console.error(e);
    }

}

/**
 * 合計値の組み合わせがtargetSumArrと一致している数を返す
 */
const exhaustiveSearch = (targetSumArr: number[], targetArr: number[], endIdx: number): void => {
    
    const sumResults: number[] = [];
    for(let i = 0; i < targetArr.length; i++){
        sumResults.push(...combination(targetArr, i, [],endIdx));
    }
    // 自分自身の要素も追加
    sumResults.push(...targetArr);
    for(let i = 0; i < targetSumArr.length; i++){
        if(sumResults.includes(targetSumArr[i])){
            console.log("yes");
        }else{
            console.log("no");
        }
    }
}

/**
 * 一方向で和の組み合わせを計算する
 */
const combination = (arr: number[], startIdx: number, results: number[],endIdx: number): number[]  =>{
    if(startIdx >= arr.length - 1){
        return results;
    }else{
        for(let i = startIdx + 1; i < endIdx; i++){
            results.push(arr[startIdx] + arr[i]);
        }
        return combination(arr, startIdx + 1, results, endIdx);
    }
}
/**
 * 配列の要素数が正しいかチェックする
 */
const checkCorrectArray = (arr: number[],inputIndex: number):void => {
    if(arr.length !== inputIndex){
        throw new Error(`配列の要素数が正しくありません。`);
    }
}
main();



/**
 * 階乗を計算する再帰関数
 * @param n 
 * @returns 
 */
const factorial  = (n: number) :number =>{
    if (n === 1 ){
        return 1;
    }
    return n * factorial(n - 1);
}

/**
 * 分割統治法を用いて配列arrのstartからend(endは含まない)までの範囲にある要素の最大値を求めるアルゴリズム
 * memo 分割統治法とは、ある問題を２つ以上の小さい問題に分割して、再帰関数によりそれぞれの部分の解を求め、
 * それらの解を組み合わせて元の問題の解を求める手法
 * @param arr　検索対象の配列
 * @param start　開始位置
 * @param end　終了位置
 * @returns 指定範囲の最大値
 */
const findMaxNum = (arr: number[], start: number, end: number): number =>{

    //case: arr= [2, 3, 4, 5, 6], start=0, end=5

    //m = 3
    const m: number = Math.floor((start + end) / 2);//中間の位置を計算する（小数点以下は切り捨て） 

    if(start === end - 1){
        return arr[start];
    }else{
        const leftMax: number = findMaxNum(arr,start,m);//左側の最大値 leftMax=3
        const rightMax: number = findMaxNum(arr,m,end);//右側の最大値  rightMax=6
        return Math.max(leftMax, rightMax); // return Math.max(3, 6) = 6
    }   
}
